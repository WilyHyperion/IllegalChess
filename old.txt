co = cohere.Client(api_key=os.environ.get("apikey"))
prompt = f"""
     play a game of chess with me. Respond only with what you move and what place it moved to Moves from this point have been: white e4, Black e5, white. Your move is:
"""
response = co.generate(  
    model='command-xlarge-nightly',  
    prompt = prompt,  
    max_tokens=40,  
    temperature=0.6)
move = response.generations[0].text

print(move)
